  def CalCrc16(self,bytestr):
        '''
        crc16查表法
        :param bytestr: bytes字符串
        :return: int16类型
        '''
        crc = 0x0000
        data = bytearray(bytestr)
        len1 = len(bytestr)
        print(len1)
        for i in range(len1):
            cp = ((crc >> 8) & 0xFF) ^ data[i]
            crc = ((crc << 8) & 0xFFFF) ^ self.g_crc16Table[cp]
        return crc
