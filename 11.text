const handleGasName = (data) => {
   let match = data.match(/-(\w+)-(\d+(-\d+)?\s?(SLM|SCCM))-/);
   if (match) {
      return [match[1], match[2].replace(/\s?(SLM|SCCM)/, '')];
   }
   return [];
}
