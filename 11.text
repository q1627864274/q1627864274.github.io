    convertDate(dateString) {
      // 将输入的日期字符串转换为 Date 对象
      const date = new Date(dateString.replace(' ', 'T') + 'Z'); // 添加 'T' 和 'Z' 来指定为 UTC 时间

      // 转换为中国时区（UTC+8）
      const offset = 8; // 中国标准时间偏移量
      const chinaTime = new Date(date.getTime() + offset * 3600 * 1000);

      // 构建 ISO 字符串
      return chinaTime.toISOString().replace('.000Z', `+08:00`);
    }
    这个函数只能转化中国时区，这个软件用在欧洲或者其他地方时区不一样，
