const { proxy } = getCurrentInstance();
const sendPortData = (writeData, callback) => {
   proxy.$SerialPortManager.write(writeData, res => {
      if (res) {
         callback(res);
      } else {
         HuiMessage({
            message: `${res}`,
            duration: 3000,
            type: 'error'
         })
      }
   });
}

// 下载下发信息
const downloadInfo = () => {
   return new Promise((resolve, reject) => {
      writeData.value.commandId = '0x0B'
      writeData.value.subCommandId = '0x01'
      writeData.value.data.push('1')
      writeData.value.data.push(...convertTo16(totalFrames.value))
      writeData.value.data.push(...convertTo16('512'))
      writeData.value.data.push(...convertTo16(crc.value))
      sendPortData(writeData.value, res => {
         let { data, commandId, subCommandId, compCode } = res
         console.log('res', res)
         if (data.length === 0 && commandId === '0b' && subCommandId === '01') {
            resolve(true)
         } else if (commandId === '0b' && (subCommandId === '02' || subCommandId === '03')) {
            // MFC请求帧数
            let resFrameNumber = parseInt(data[0] + compCode, 16)
            console.log('downloadInfo里面的resFrameNumber帧数', resFrameNumber)
            let sliceData = bytArray.value.slice(resFrameNumber * 512, (resFrameNumber + 1) * 512)
            let sliceDataCopy = []
            sliceData.forEach(item => {
               sliceDataCopy.push(Number(item).toString(16))
            })
            downloadDataItem(resFrameNumber, sliceDataCopy)
         }
      })
   })
}
