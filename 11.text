   <script>
      let redTime = 3; // 红灯时间
      let yellowTime = 1; // 黄灯时间
      let greenTime = 3; // 绿灯时间

      function changeLight() {
         document.getElementById('red').style.backgroundColor = "red";
         document.querySelector('.time').textContent = redTime;
         setTimeout(() => {
            document.getElementById('red').style.backgroundColor = "grey";
            document.getElementById('yellow').style.backgroundColor = "yellow";
            document.querySelector('.time').textContent = yellowTime;
            setTimeout(() => {
               document.getElementById('yellow').style.backgroundColor = "grey";
               document.getElementById('green').style.backgroundColor = "green";
               document.querySelector('.time').textContent = greenTime;
               setTimeout(() => {
                  document.getElementById('green').style.backgroundColor = "grey";
                  changeLight(); // 重新开始循环
               }, greenTime * 1000);
            }, yellowTime * 1000);
         }, redTime * 1000);
      }
   
      changeLight(); // 开始灯光切换循环
      // 这种写法，<p>时间: <span class="time">0</span> 秒</p>这个标签的时间是静态，希望呈现3 2,1这样倒计时，然后切换灯之后再倒计时，怎么实现？
